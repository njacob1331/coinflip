struct Cache<T> {
    inner: Option<T>,
}

impl<T> Cache<T> {
    fn new() -> Self {
        Self { inner: None }
    }

    fn with_init(value: T) -> Self {
        Self { inner: Some(value) }
    }

    fn read(&self) -> Option<&T> {
        self.inner.as_ref()
    }

    fn read_mut(&mut self) -> Option<&mut T> {
        self.inner.as_mut()
    }

    fn write(&mut self, value: T) {
        self.inner = Some(value)
    }

    fn write_if<P>(&mut self, value: T, predicate: P) -> bool
    where
        P: FnOnce(&T, &T) -> bool,
    {
        let Some(inner) = self.inner.as_ref() else {
            self.write(value);
            return true;
        };

        if predicate(inner, &value) {
            self.write(value);
            true
        } else {
            false
        }
    }

    fn is_empty(&self) -> bool {
        self.inner.is_none()
    }

    fn clear(&mut self) -> Option<T> {
        self.inner.take()
    }
}

use std::{
    collections::{HashMap, hash_map::Entry},
    hash::Hash,
    sync::{
        Arc,
        atomic::{AtomicU32, Ordering},
    },
};

#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
pub struct GlobalId(u32);

impl GlobalId {
    const SOURCE_BITS: u32 = 2;
    const LOCAL_BITS: u32 = 32 - Self::SOURCE_BITS;
    const MAX_SOURCE_ID: u32 = (1 << Self::SOURCE_BITS) - 1;
    const MAX_LOCAL_ID: u32 = (1 << Self::LOCAL_BITS) - 1;
    const LOCAL_MASK: u32 = Self::MAX_LOCAL_ID;

    fn new(source_id: u32, local_id: u32) -> Self {
        Self((source_id << Self::LOCAL_BITS) | local_id)
    }

    fn source_id(self) -> u32 {
        self.0 >> Self::LOCAL_BITS
    }

    fn local_id(self) -> u32 {
        self.0 & Self::LOCAL_MASK
    }
}

#[derive(Debug)]
pub struct Interned<T> {
    pub id: GlobalId,
    pub inner: T,
}

#[derive(Debug)]
struct Interner<K> {
    id: u32,
    index: HashMap<K, GlobalId>,
    reverse: Vec<Option<K>>,
}

impl<K> Interner<K>
where
    K: Clone + Hash + Eq,
{
    fn new(id_generator: Arc<AtomicU32>) -> Self {
        let id = id_generator.fetch_add(1, Ordering::Relaxed);

        Self {
            id,
            index: HashMap::new(),
            reverse: Vec::new(),
        }
    }

    fn intern(&mut self, key: K) -> GlobalId {
        match self.index.entry(key) {
            Entry::Occupied(entry) => *entry.get(),
            Entry::Vacant(entry) => {
                let local_id = self.reverse.len() as u32;
                let global_id = GlobalId::new(self.id, local_id);

                self.reverse.push(Some(entry.key().clone()));
                entry.insert(global_id);

                global_id
            }
        }
    }

    fn lookup(&self, id: GlobalId) -> Option<&K> {
        if id.source_id() != self.id {
            return None;
        }

        self.reverse.get(id.local_id() as usize)?.as_ref()
    }

    fn remove(&mut self, key: &K) -> Option<K> {
        let id = self.index.remove(key)?;

        if id.source_id() != self.id {
            return None;
        }

        self.reverse.get_mut(id.local_id() as usize)?.take()
    }

    fn wrap<T>(&mut self, key: K, data: T) -> Interned<T> {
        let id = self.intern(key);

        Interned { id, inner: data }
    }
}

use std::{borrow::Borrow, collections::HashMap, hash::Hash};

#[derive(Debug, PartialEq, Eq)]
pub struct Observation<T> {
    pub ts: u64,
    pub value: T,
}

#[derive(Debug, PartialEq, Eq, Clone, Copy)]
pub enum WindowOrdering {
    Within,
    Before,
    After,
}

#[derive(Debug)]
pub struct Window<K, V> {
    start: u64,
    end: u64,
    observations: HashMap<K, Vec<Observation<V>>>,
}

impl<K, V> Window<K, V>
where
    K: Hash + Eq,
{
    fn new(start: u64, end: u64) -> Self {
        Self {
            start,
            end,
            observations: HashMap::new(),
        }
    }

    fn cmp_obs(&self, obs: &Observation<V>) -> WindowOrdering {
        if obs.ts < self.start {
            WindowOrdering::Before
        } else if obs.ts >= self.end {
            WindowOrdering::After
        } else {
            WindowOrdering::Within
        }
    }

    fn push(&mut self, id: K, obs: Observation<V>) {
        debug_assert!(self.start <= obs.ts && obs.ts < self.end);

        let observations = self.observations.entry(id).or_default();

        if observations.last().is_none_or(|last| obs.ts >= last.ts) {
            observations.push(obs);
            return;
        }

        let index = observations.partition_point(|existing| existing.ts <= obs.ts);

        observations.insert(index, obs);
    }
}

#[derive(Debug)]
pub struct WindowView<'a, K, V> {
    window: Option<&'a Window<K, V>>,
}

impl<'a, K, V> WindowView<'a, K, V>
where
    K: Hash + Eq,
{
    pub fn query(self, id: &K) -> Query<impl Iterator<Item = &'a Observation<V>>> {
        self.window
            .and_then(|window| window.observations.get(id))
            .into_iter()
            .flatten()
            .into()
    }

    pub fn query_multiple<I>(self, ids: I) -> Query<impl Iterator<Item = &'a Observation<V>>>
    where
        I: IntoIterator,
        I::Item: Borrow<K>,
    {
        let window = self.window;

        ids.into_iter()
            .flat_map(move |id| {
                window
                    .and_then(|window| window.observations.get(id.borrow()))
                    .into_iter()
                    .flatten()
            })
            .into()
    }
}

pub struct Query<I> {
    iter: I,
}

impl<I> From<I> for Query<I>
where
    I: Iterator,
{
    fn from(value: I) -> Self {
        Self { iter: value }
    }
}

impl<I> Iterator for Query<I>
where
    I: Iterator,
{
    type Item = I::Item;

    fn next(&mut self) -> Option<Self::Item> {
        self.iter.next()
    }
}

impl<'a, I, V> Query<I>
where
    I: Iterator<Item = &'a Observation<V>>,
    V: 'a,
{
    pub fn values(self) -> impl Iterator<Item = &'a V> {
        self.map(|obs| &obs.value)
    }
}

#[derive(Debug, PartialEq, Eq)]
pub enum LogResult<T> {
    Success,
    Rejected(T),
}

impl<T> LogResult<T> {
    fn success(&self) -> bool {
        matches!(self, Self::Success)
    }

    fn rejected(&self) -> bool {
        matches!(self, Self::Rejected(_))
    }
}

#[derive(Debug, Default, Clone, Copy)]
pub enum LogBehavior {
    StrictlySequential,
    #[default]
    CurrentOrFuture,
    Lax,
}

#[derive(Debug)]
pub struct TimeSeries<K, V> {
    windows: Vec<Window<K, V>>,
    window_size_ms: u64,
    log_behavior: LogBehavior,
}

impl<K, V> TimeSeries<K, V>
where
    K: Hash + Eq,
{
    pub fn new(window_size_ms: u64, log_behavior: LogBehavior) -> Self {
        Self {
            windows: Vec::new(),
            window_size_ms,
            log_behavior,
        }
    }

    pub fn num_windows(&self) -> usize {
        self.windows.len()
    }

    /// Returns a view into the current window of the timeseries if one exists
    pub fn current(&self) -> WindowView<'_, K, V> {
        WindowView {
            window: self.windows.last(),
        }
    }

    pub fn last_n(&self, n: usize) {
        let n = n.saturating_sub(1);
        let last = self.num_windows().saturating_sub(1);

        let _slice = self.windows.get(n..last);
    }

    fn bucket_start(&self, ts: u64) -> u64 {
        (ts / self.window_size_ms) * self.window_size_ms
    }

    fn bucket_end(&self, start: u64) -> u64 {
        start
            .checked_add(self.window_size_ms)
            .expect("window end timestamp overflowed")
    }

    /// transforms the value in place with the provided closure then logs
    pub fn transform_and_log<T>(
        &mut self,
        id: K,
        mut obs: Observation<V>,
        transform: T,
    ) -> LogResult<Observation<V>>
    where
        T: FnOnce(&mut V),
    {
        transform(&mut obs.value);
        self.log(id, obs)
    }

    pub fn log(&mut self, id: K, obs: Observation<V>) -> LogResult<Observation<V>> {
        let start = self.bucket_start(obs.ts);

        match self.log_behavior {
            LogBehavior::CurrentOrFuture => match self.windows.last_mut() {
                Some(current) => match current.cmp_obs(&obs) {
                    WindowOrdering::Within => {
                        current.push(id, obs);
                        LogResult::Success
                    }
                    WindowOrdering::Before => LogResult::Rejected(obs),
                    WindowOrdering::After => {
                        let end = self.bucket_end(start);
                        let mut window = Window::new(start, end);
                        window.push(id, obs);
                        self.windows.push(window);
                        LogResult::Success
                    }
                },
                None => {
                    let end = self.bucket_end(start);

                    let mut window = Window::new(start, end);
                    window.push(id, obs);

                    self.windows.push(window);

                    LogResult::Success
                }
            },

            LogBehavior::StrictlySequential => {
                match self.windows.last().map(|window| window.start) {
                    None => {
                        let end = self.bucket_end(start);

                        let mut window = Window::new(start, end);
                        window.push(id, obs);

                        self.windows.push(window);

                        LogResult::Success
                    }
                    Some(current_start) if start < current_start => LogResult::Rejected(obs),
                    Some(current_start) if start == current_start => {
                        let index = self.windows.len() - 1;
                        self.windows[index].push(id, obs);

                        LogResult::Success
                    }
                    Some(current_start) => {
                        let next_start = current_start
                            .checked_add(self.window_size_ms)
                            .expect("window start timestamp overflowed");

                        if start == next_start {
                            let end = self.bucket_end(start);

                            let mut window = Window::new(start, end);
                            window.push(id, obs);

                            self.windows.push(window);

                            LogResult::Success
                        } else {
                            LogResult::Rejected(obs)
                        }
                    }
                }
            }

            LogBehavior::Lax => {
                let end = self.bucket_end(start);

                match self
                    .windows
                    .binary_search_by_key(&start, |window| window.start)
                {
                    Ok(index) => {
                        self.windows[index].push(id, obs);
                    }
                    Err(index) => {
                        let mut window = Window::new(start, end);
                        window.push(id, obs);

                        self.windows.insert(index, window);
                    }
                }

                LogResult::Success
            }
        }
    }
}

    Millis,

#[derive(Debug, PartialEq, Eq, Clone, Copy)]
pub enum WindowOrdering {
    Within,
    Before,
    After,
}

#[derive(Debug)]
pub struct Window<K, V> {
    start: u64,
    end: u64,
    observations: HashMap<K, Vec<V>>,
}

impl<K, V> Window<K, V>
where
    K: Hash + Eq,
    V: Observation,
{
    fn new(start: u64, end: u64) -> Self {
        Self {
            start,
            end,
            observations: HashMap::new(),
        }
    }

    fn cmp_obs(&self, obs: &V) -> WindowOrdering {
        let timestamp = obs.timestamp();

        if timestamp < self.start {
            WindowOrdering::Before
        } else if timestamp >= self.end {
            WindowOrdering::After
        } else {
            WindowOrdering::Within
        }
    }

    /// Pushes an item into the window categorized by id.
    /// This method does not enforce strict ordering.
    fn push(&mut self, id: K, obs: V) {
        self.observations.entry(id).or_default().push(obs);
    }

    /// Pushes an item into the window categorized by id ONLY IF the provided observation is
    /// the same time as or after the last recorded observation.
    fn try_push(&mut self, id: K, obs: V) -> Result<(), V> {
        let observations = self.observations.entry(id).or_default();

        if observations
            .last()
            .is_some_and(|last| obs.timestamp() < last.timestamp())
        {
            return Err(obs);
        }

        observations.push(obs);
        Ok(())
    }
}

/// Controls how tolerant the time series is to out-of-order observations.
/// By default, the series will tolerate out-of-order observations as long
/// as they fall within the current window.
#[derive(Debug, Default, Clone, Copy)]
pub enum Tolerance {
    None,
    #[default]
    WithinWindow,
    AcrossWindows,
}

#[derive(Debug)]
pub struct TimeSeries<K, V> {
    windows: Vec<Window<K, V>>,
    window_size_ms: u64,
    tolerance: Tolerance,
}

impl<K, V> TimeSeries<K, V>
where
    K: Hash + Eq,
    V: Observation,
{
    pub fn new(window_size_ms: u64, tolerance: Tolerance) -> Self {
        Self {
            windows: Vec::new(),
            window_size_ms,
            tolerance,
        }
    }

    pub fn num_windows(&self) -> usize {
        self.windows.len()
    }

    /// Returns a view into the current window of the timeseries if one exists
    pub fn current(&self) -> WindowView<'_, K, V> {
        WindowView {
            window: self.windows.last(),
        }
    }

    #[inline]
    fn compute_bounds(&self, obs: &V) -> (u64, u64) {
        let lower_bounds = (obs.timestamp() / self.window_size_ms) * self.window_size_ms;
        let upper_bounds = lower_bounds + self.window_size_ms;

        (lower_bounds, upper_bounds)
    }

    fn create_window(&mut self, id: K, obs: V) {
        let (start, end) = self.compute_bounds(&obs);

        let mut window = Window::new(start, end);
        window.push(id, obs);

        self.windows.push(window);
    }

    /// transforms the value in place with the provided closure then logs
    pub fn transform_and_record<T>(&mut self, id: K, mut obs: V, transform: T) -> Result<(), V>
    where
        T: FnOnce(&mut V),
    {
        transform(&mut obs);
        self.record(id, obs)
    }

    pub fn record(&mut self, id: K, obs: V) -> Result<(), V> {
        match self.tolerance {
            Tolerance::None => {
                let Some(current) = self.windows.last_mut() else {
                    self.create_window(id, obs);
                    return Ok(());
                };

                match current.cmp_obs(&obs) {
                    WindowOrdering::Within => current.try_push(id, obs),
                    WindowOrdering::After => {
                        self.create_window(id, obs);
                        Ok(())
                    }
                    WindowOrdering::Before => Err(obs),
                }
            }

            Tolerance::WithinWindow => {
                let Some(current) = self.windows.last_mut() else {
                    self.create_window(id, obs);
                    return Ok(());
                };

                match current.cmp_obs(&obs) {
                    WindowOrdering::Within => current.push(id, obs),
                    WindowOrdering::After => self.create_window(id, obs),
                    WindowOrdering::Before => return Err(obs),
                }

                Ok(())
            }

            Tolerance::AcrossWindows => {
                let (start, end) = self.compute_bounds(&obs);

                match self
                    .windows
                    .binary_search_by_key(&start, |window| window.start)
                {
                    Ok(index) => {
                        self.windows[index].push(id, obs);
                    }
                    Err(index) => {
                        let mut window = Window::new(start, end);
                        window.push(id, obs);

                        self.windows.insert(index, window);
                    }
                }

                Ok(())
            }
        }
    }
}

#[cfg(test)]
mod tests {
    use std::time::{Duration, SystemTime, UNIX_EPOCH};

    use crate::{Observation, TimeSeries, Tolerance};

    fn ts() -> u64 {
        SystemTime::now()
            .duration_since(UNIX_EPOCH)
            .unwrap()
            .as_millis() as u64
    }

    #[derive(Observation, Debug, Clone, Copy)]
    #[observation(timestamp = ts, fields(temp, humidity))]
    struct SensorReading {
        ts: u64,
        temp: f64,
        humidity: f64,
    }

    #[test]
    fn init() {
        let (tx, rx) = std::sync::mpsc::channel();
        let mut series = TimeSeries::new(10, Tolerance::default());

        std::thread::spawn(move || {
            for _ in 0..100 {
                tx.send(SensorReading {
                    ts: ts(),
                    temp: 88.0,
                    humidity: 1.0,
                })
                .unwrap();

                std::thread::sleep(Duration::from_millis(1));
            }
        });

        while let Ok(reading) = rx.recv() {
            // this method may fail based on our tolerance
            // in the case, we know it won't so we unwrap
            series.record("Atlanta", reading).unwrap()
        }

        println!("{series:#?}")
    }

    #[test]
    fn tolerance() {
        let one = SensorReading {
            ts: 0,
            temp: 87.0,
            humidity: 2.0,
        };

        let two = SensorReading {
            ts: 1,
            temp: 88.0,
            humidity: 2.0,
        };

        let three = SensorReading {
            ts: 0,
            temp: 87.2,
            humidity: 2.0,
        };

        let mut default = TimeSeries::new(1, Tolerance::default());
        assert!(default.record("Atlanta", one).is_ok());
        // new window created here
        assert!(default.record("Atlanta", two).is_ok());
        assert!(default.record("Atlanta", three).is_err());

        let mut lax = TimeSeries::new(1, Tolerance::AcrossWindows);
        assert!(lax.record("Atlanta", one).is_ok());
        // new window created here
        assert!(lax.record("Atlanta", two).is_ok());
        assert!(lax.record("Atlanta", three).is_ok());
    }
